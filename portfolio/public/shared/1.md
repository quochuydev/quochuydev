title: Raspberry Pi 4 System Monitoring Commands

---

Disk Space

# Overall disk usage

df -h

# Disk usage of current directory

du -sh \*

# Find largest directories from root

sudo du -h --max-depth=1 / 2>/dev/null | sort -hr | head -20

# Find largest files (>100MB)

sudo find / -type f -size +100M -exec ls -lh {} \; 2>/dev/null | sort -k5 -hr

# Check n8n data directory size

du -sh /home/n8n/.n8n

Memory Usage

# Memory overview

free -h

# Detailed memory info

cat /proc/meminfo

# Top memory-consuming processes

ps aux --sort=-%mem | head -15

# Interactive view

htop

CPU Usage

# Top CPU-consuming processes

ps aux --sort=-%cpu | head -15

# Real-time CPU per core

htop

# CPU temperature

vcgencmd measure_temp

Package/App Sizes

# List installed packages by size

dpkg-query -Wf '${Installed-Size}\t${Package}\n' | sort -rn | head -30

# With human-readable sizes

dpkg-query -Wf '${Installed-Size}\t${Package}\n' | sort -rn | head -30 | awk '{printf "%.2f MB\t%s\n", $1/1024, $2}'

# List npm global packages

npm list -g --depth=0

# Size of npm global packages

du -sh /usr/lib/node_modules/\*

All-in-One System Check

# Quick system overview

echo "=== DISK ===" && df -h / && echo -e "\n=== MEMORY ===" && free -h && echo -e "\n=== CPU TEMP ===" && vcgencmd measure_temp && echo -e "\n=== TOP PROCESSES ===" && ps aux --sort=-%mem | head -6

Install Useful Tools

# htop - interactive process viewer

sudo apt install htop

# ncdu - interactive disk usage analyzer

sudo apt install ncdu
sudo ncdu / # Navigate directories interactively

# duf - modern df alternative

sudo apt install duf
duf
